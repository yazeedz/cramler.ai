## API Calls

### Step 1 â†’ Step 2
```typescript
// Create organization
const { data: org } = await supabase
  .from('organizations')
  .insert({ name, slug, industry, website })
  .select()
  .single();

// Add user as owner
await supabase
  .from('organization_members')
  .insert({
    organization_id: org.id,
    user_id: user.id,
    role: 'owner',
    accepted_at: new Date().toISOString(),
  });
```

### Step 2 â†’ Step 3
```typescript
// Create brand
const { data: brand } = await supabase
  .from('brands')
  .insert({
    organization_id: orgId,
    name,
    slug,
    category_id,
    market_position,
    website,
  })
  .select()
  .single();
```

### Step 3 â†’ Dashboard
```typescript
// Create products
for (const productName of products) {
  // Insert product
  const { data: product } = await supabase
    .from('products')
    .insert({
      brand_id: brandId,
      name: productName,
      slug: generateSlug(productName),
      status: 'pending',
      created_by: user.id,
    })
    .select()
    .single();

  // Trigger AI research
  await fetch('/api/n8n-webhook', {
    method: 'POST',
    body: JSON.stringify({
      product_id: product.id,
      product_name: productName,
      user_id: user.id,
    }),
  });
}

// Redirect to dashboard
router.push('/dashboard');
```
